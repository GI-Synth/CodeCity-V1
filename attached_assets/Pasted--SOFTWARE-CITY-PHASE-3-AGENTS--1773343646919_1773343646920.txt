class TestExecutor {
async executeTests(params: {
 targetFile: string, // absolute path to file being tested
 testCode: string, // generated test code as string
 language: string, // "typescript" | "javascript" | "python"
 timeoutMs?: number, // default 15000
}): Promise<TestExecutionResult>
IMPLEMENTATION:
 1. Write testCode to a temp file:
 /tmp/sc_test_[uuid].[ts|js|py]
 Prepend the correct import for the target file
 2. For TypeScript/JavaScript:
 Run: npx c8 --reporter=json-summary tsx [tempFile]
 Use child_process.spawn with timeout
 Parse stdout for test results (look for PASS/FAIL lines)
 Parse coverage-summary.json if it exists in ./coverage/
 3. For Python:
 Run: python -m pytest [tempFile] --json-report
 --json-report-file=/tmp/sc_pytest_[uuid].json
 Parse the JSON report file for results
 4. Parse stdout/stderr for any thrown errors → these are bugs
 Capture: error message, line number, stack trace
 5. Clean up temp files always (try/finally)
 6. Return TestExecutionResult:
 {
 passed: number,
 failed: number,
 errors: TestError[], // { message, line, stack }
 coverage: number, // 0.0-1.0, null if not available
 durationMs: number,
 rawOutput: string, // truncated to 2000 chars
 }
async runQuickSanity(filePath: string): Promise<SanityResult>
 Just try to import/require the file and catch any syntax errors
 Run: node --input-type=module <<< "import './[file]'"
 Returns { ok: boolean, error?: string }
}
parseTestOutput(stdout: string, stderr: string): ParsedResults Look for patterns: “X passing” / “X failing” (Mocha style) “Tests: X
passed, Y failed” (Jest style) “PASSED” / “FAILED” per line (pytest style) Any “Error:” or “TypeError:” or “AssertionError:” in stderr
parseCoverageJSON(jsonPath: string): number Read coverage-summary.json Return total.lines.pct / 100 as 0.0-1.0 Return null if file
doesn’t exist
